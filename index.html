const FOLDER_ID = '1pZN6e7oH-kMGOs8lNwWX2fCTf_RTQBvc';

function doGet(e) {
  try {
    const folder = DriveApp.getFolderById(FOLDER_ID);
    const files = folder.getFilesByType('application/pdf');
    const result = [];
    while (files.hasNext()) {
      const f = files.next();
      const name = f.getName();
      const licKey = extractKey(name);
      result.push({ name: name, id: f.getId(), licKey: licKey,
        url: 'https://drive.google.com/file/d/' + f.getId() + '/view' });
    }
    return ContentService
      .createTextOutput(JSON.stringify({ files: result, count: result.length }))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ error: err.message }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function extractKey(filename) {
  const name = filename.replace(/\.pdf$/i, '').trim();
  const m1 = name.match(/(\d{2}-\d{1,2}-\d{1,2}-\d{1,2}-\d{7})/);
  if (m1) return m1[1];
  const m2 = name.match(/\b(\d{7})\b/);
  if (m2) return m2[1];
  return name;
}
